14// Boundary condition types

16enum class BCType : int32_t {
17 INTERIOR = 0, // Interior edge — not a boundary
18 WALL = 1, // Solid wall: zero normal flux
19 INFLOW_Q = 2, // Prescribed unit discharge normal to edge (m²/s)
20 STAGE = 3, // Prescribed water-surface elevation (m)
21 OPEN = 4, // Riemann outflow / zero-gradient
22 REFLECT = 5, // Reflecting: reverse normal velocity component
23 NORMAL_DEPTH = 6, // Prescribed boundary depth h (m)

43 std::vector<int32_t> cell_face_nodes; // [sum(n_verts_cell)] node indices for each cell ring
44 std::vector<int32_t> cell_edge_offsets; // [n_cells + 1], CSR offsets into cell_edge_ids

63 std::vector<double> edge_bc_val; // [n_edges] prescribed BC value (context-dependent)
64
65 // ── Derived per-cell geometry (built once, used by solver) ───────────────
66 // For each cell c, inverse area is stored for the update step.
67 std::vector<double> cell_inv_area; // [n_cells] 1/area (m⁻²)
68
69 // ── 2-ring cell stencil (CSR) — used by least-squares gradient (scheme 6) ─
70 // For each cell, the set of unique neighbour cells reachable via 1 or 2
71 // interior-edge traversals (excluding self). Precomputed Δx/Δy to each
72 // neighbour centroid and inverse-distance² weights feed the LSQ gradient.
73 std::vector<int32_t> cell_ring2_offsets; // [n_cells + 1] CSR offsets
74 std::vector<int32_t> cell_ring2_ids; // [sum(ring2_counts)] neighbour cell indices
75 std::vector<double> cell_ring2_dcx; // [sum(ring2_counts)] Δx = cx[j]-cx[c]
76 std::vector<double> cell_ring2_dcy; // [sum(ring2_counts)] Δy = cy[j]-cy[c]
77 std::vector<double> cell_ring2_inv_dist2; // [sum(ring2_counts)] 1/|Δr|² weight
