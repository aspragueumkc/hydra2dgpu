156inline bool swe2d_contract_is_valid(const SWE2D3DInterfaceContractHost& c) {
157 if (c.cell2d.empty()) return false;
158 const size_t n = c.cell2d.size();
159 return c.face_area.size() == n &&
160 c.face_nx.size() == n &&
161 c.face_ny.size() == n &&
162 c.face_nz.size() == n;
163}
